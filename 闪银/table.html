var aData = [
    [1, 2, 33, 5],
    [2, 10, 15, 16],
    [3, 1, 21, 20]
];

/**
 * 添加表格
 * @param array
 * @param htmlTagId 父标签ID
 */

var fnAddTable = (array = [], htmlTagId = 'answer') => {
    var hParent = document.getElementById(htmlTagId);

    if (!Array.isArray(array) || !hParent) {

        var sErrorTxt = Array.isArray(array) ? hParent ? '' : '找不到标签' : '数据类型错误';
        console.log(sErrorTxt);

        return
    }

    var hTable = document.createElement('table');

    array.forEach((arrayEle) => {

        if (!Array.isArray(arrayEle)) {
            var sErrorTxt = '数据类型错误';
            console.log(sErrorTxt);
            return
        }

        var hTr = document.createElement('tr'),
            maxNum = Math.max ? Math.max.apply(this, arrayEle) : null; //取出每行数据的最大值

        arrayEle.forEach((ele) => {

            var hTd = document.createElement('td');

            hTd.innerHTML = ele;

            hTd.className = maxNum === ele ? "fontRed" : '';

            hTr.appendChild(hTd)
        });

        hTable.appendChild(hTr);

    });

    hParent.appendChild(hTable)
};

fnAddTable(aData, 'answer')
